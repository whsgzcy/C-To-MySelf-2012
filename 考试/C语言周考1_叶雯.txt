29. 求两个正整数的最大公约数和最小公倍数。
#include <stdio.h>

int max(int num1,int num2)
{
	int i;
	int res = 1;
	for(i = 2; i <= num1; i++)
	{
		if(num1 % i == 0 && num2 % i == 0)
		{
			res = i;
		}
	}
	return res;
}


int min(int num1,int num2)
{
	int i;
	int res = num1 * num2;
	for(i = num1 * num2; i >= num1; i--)
	{
		if(i % num1 == 0 && i % num2 == 0)
			res = i;
	}
		return res;
}

	
void main()
{
	int x,y;
	printf("输入两个数:");
	scanf("%d,%d",&x,&y);
	printf("%d and %d 最大公约数 %d\n ",x,y,max(x,y));
	printf("%d and %d 最小公约数 %d\n ",x,y,min(x,y));
}
