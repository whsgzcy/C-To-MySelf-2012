#include <stdio.h>//二维数组转置

void main()
{
	int a[3][4] = {{1, 2, 3, 4}, {5, 6, 7, 8}, {9, 1, 2, 3}};
	int b[4][3];
	int i, j;

	for(i = 0; i < 3; i++)
	{
		for(j = 0; j < 4; j++)
		{
			printf("%2d ", a[i][j]);
			b[j][i] = a[i][j];
		}
			printf("\n");
	}
	printf("\n");
	for(i = 0; i < 4; i++)
	{
		for(j = 0; j < 3; j++)
		{
			printf("%d ", b[i][j]);
		}
		printf("\n");
	}
}
