int fun(int a[10])
{
	int i;

	for(i = 0; i < 10; i++)
	{
		if(a[0] != a[i])
		{
			return 1;
		}
	}
	return 0;
}

void print(int a[10])
{
	int i;

	for(i = 0; i < 10; i++)
	{
		printf("%d ", a[i]);
	}
	printf("\n");
}

void main()
{
	void print(int a[10]);
	int fun(int a[10]);
	int a[10] = {10, 2, 8, 22, 16, 4, 10, 6, 14, 20};
	int i;
	int b[10];

	while(fun(a))
	{
		for(i = 0; i < 10; i++)
		{
			if(a[i] % 2 == 0)
				b[i] = a[i] = a[i] / 2;
			else
				b[i] = a[i] = (a[i] + 1) / 2;
		}

		for(i = 0; i < 9; i++)
		{
			a[i + 1] = a[i + 1] + b[i];
		}
		a[0] = a[0] + b[9];

		print(a);
	}
		printf("\n");
}
