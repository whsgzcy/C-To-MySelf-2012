#include <stdio.h>
#include <math.h>

void main()
{
	int m, k, i, n = 0;

	for(m = 1;m <= 100;m++)//m++?这里可以优化为m = m + 2;
	{
		k = sqrt(m);//???此时的开平方
		for(i = 2;i <= k;i++)

			if(m % i == 0)
				break;//终止全部循环
			if(i >= k + 1)
			{
				printf("%d ", m);
				n = n + 1;
			}
	}	
}
